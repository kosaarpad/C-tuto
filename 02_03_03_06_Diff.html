C# középiskolásoknak

Példaprogram: halmazok különbsége

Adott két halmaz: az A és a B.
A = {1,2,3,4,5}
B = {3,4,5,6,7,8,9}

1. Feladat: képezzük az A és B halmazok külünbségét, a $C = A \setminus B$ halmazt:
FIGYELEM!!! Általában $A \setminus B\neq B \setminus A$

using System;
using System.Collections.Generic;
class Program
	{
	private static void HalmazMutat(HashSet<int> halmaz)
		{
		Console.Write("{");
		foreach (int elem in halmaz)
			{
			Console.Write(" {0}", elem);
			}
		Console.WriteLine(" }");
		}
	static void Main()
		{
		HashSet<int> A = new HashSet<int>(){1,2,3,4,5};//itt a halmazokat tömbfelsiorolásból képezzük
		HashSet<int> B = new HashSet<int>(){3,4,5,6,7,8,9};
		Console.ForegroundColor = ConsoleColor.Yellow;
		Console.WriteLine("Az A halmaz:");
		HalmazMutat(A);
		Console.WriteLine("A B halmaz:");
		HalmazMutat(B);
		//halmazok különbsége:  
		HashSet<int> C = new HashSet<int>(A);
		C.ExceptWith(B);
		Console.WriteLine();
		Console.WriteLine(" A és B különbsége a C:");
		HalmazMutat(C);
		}
	}
